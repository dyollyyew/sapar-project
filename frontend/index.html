async function search() { const resDiv = document.getElementById('res'); const from = document.getElementById('from').value.trim().toUpperCase(); const to = document.getElementById('to').value.trim().toUpperCase(); const date = document.getElementById('date').value; if (!date) return alert("Пожалуйста, выберите дату!"); resDiv.innerHTML = "🔍 Ищем билеты..."; try { // Если ты запускаешь файл просто с компьютера, укажи полный URL Railway здесь const response = await fetch('/api/search-live', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ origin: from, destination: to, date: date }) }); if (!response.ok) throw new Error("Сервер ответил ошибкой: " + response.status); const data = await response.json(); resDiv.innerHTML = ""; if (data.tickets && data.tickets.length > 0) { data.tickets.forEach(t => { const price = t.proposals[0].value; resDiv.innerHTML += `
${price} RUB
Рейс: ${from} → ${to}
Купить билет →
`; }); } else { resDiv.innerHTML = "☹️ Билетов не найдено. Попробуйте другую дату."; } } catch (e) { // Теперь ты увидишь РЕАЛЬНУЮ причину ошибки прямо на экране resDiv.innerHTML = `Ошибка: ${e.message}`; console.error("Full error details:", e); } }
