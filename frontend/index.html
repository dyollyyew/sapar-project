async function searchFlights() { const from = document.getElementById('fromCity').value; // Код IATA (ASB) const to = document.getElementById('toCity').value; // Код IATA (KZN) const date = document.getElementById('travelDate').value; // ГГГГ-ММ-ДД const resultsDiv = document.getElementById('flightResults'); resultsDiv.innerHTML = '

Ищем лучшие варианты...

'; try { // Запрос идет к твоему серверу на Railway const response = await fetch(`/api/search?origin=${from}&destination=${to}&date=${date}`); const data = await response.json(); resultsDiv.innerHTML = ''; // Очистка if (data.length > 0) { data.forEach(flight => { const ticket = document.createElement('div'); ticket.className = 'ticket-card'; ticket.innerHTML = `

Цена: ${flight.value} RUB

Маршрут: ${flight.origin} → ${flight.destination}

Дата вылета: ${flight.depart_date}

Перейти к покупке
`; resultsDiv.appendChild(ticket); }); } else { resultsDiv.innerHTML = '

Билеты не найдены на эту дату.

'; } } catch (err) { resultsDiv.innerHTML = '

Произошла ошибка при поиске.

'; console.error(err); } }
